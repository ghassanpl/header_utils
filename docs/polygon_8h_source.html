106 tvec edge_point(T t, T precalced_length) const
107 {
108 auto c = vertices.size();
109 if (c < 2)
110 return c ? vertices[0] : tvec{};
111
112 t = fmod(t, precalced_length);
113 for (size_t i = 0; i < c - 1; ++i)
114 {
115 const auto d = glm::distance(vertices[i], vertices[i + 1]);
116 if (t <= d)
