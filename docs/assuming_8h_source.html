275 decltype(auto) GetFormattable(T&& val)
276 {
277 using simple_type = std::remove_cvref_t<T>;
278#if ASSUMING_INCLUDE_MAGIC_ENUM
279 if constexpr (std::is_enum_v<simple_type>)
280 return magic_enum::enum_name(val);
281 else
282#endif
283 if constexpr (std::is_constructible_v<std::string_view, simple_type>)
284 return std::forward<T>(val);
285 else if constexpr (std::is_pointer_v<simple_type>)
286 return (const void*)std::to_address(std::forward<T>(val));
287 else if constexpr (streamable<simple_type> && !formattable<simple_type>)
288 {
289 std::stringstream ss;
290 ss << std::forward<T>(val);
291 return std::move(ss).str();
292 }
293 else if constexpr (!formattable<simple_type> && !streamable<simple_type>)
294 return std::format("<{}>", typeid(simple_type).name());
295 else
296 return std::forward<T>(val);
297 }
