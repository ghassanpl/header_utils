568 template <char DELIMITER = '\''>
569 [[nodiscard]] inline std::pair<std::string_view, std::string> consume_c_string(std::string_view& strv)
570 {
571 if (strv.empty() || strv[0] != DELIMITER)
572 return {};
573
574 std::pair<std::string_view, std::string> result;
575
576 auto view = strv;
577 auto start = view.begin();
578 view.remove_prefix(1);
579 while (view[0] != DELIMITER)
580 {
581 auto cp = consume(view);
582 if (cp == '\\')
583 {
584 cp = consume(view);
585 if (view.empty())
586 return {};
587
588 switch (cp)
589 {
590 case 'n': result.second += '\n'; break;
591 case '"': result.second += '"'; break;
592 case '\'': result.second += '\''; break;
593 case '\\': result.second += '\\'; break;
594 case 'b': result.second += '\b'; break;
595 case 'r': result.second += '\r'; break;
596 case 'f': result.second += '\f'; break;
597 case 't': result.second += '\t'; break;
598 case '0': result.second += '\0'; break;
599 case 'o':
600 {
601 auto num = consume_n(view, 3);
602 if (num.size() < 3 || view.empty()) return {};
603
604 auto parsed = consume_c_integer(num, 8);
605 if (parsed.first.empty() || !num.empty()) return {};
606
607 if (parsed.second > 255) return {};
608 result.second.push_back((char)parsed.second);
609 break;
610 }
611 case 'x':
612 {
613 auto num = consume_n(view, 2);
614 if (num.size() < 2 || view.empty()) return {};
615
616 auto parsed = consume_c_integer(num, 16);
617 if (parsed.first.empty() || !num.empty()) return {};
618
619 append_utf8(result.second, (char32_t)parsed.second);
620 break;
621 }
622 case 'u':
623 {
624 auto num = consume_n(view, 4);
625 if (num.size() < 4 || view.empty()) return {};
626
627 auto parsed = consume_c_integer(num, 16);
628 if (parsed.first.empty() || !num.empty()) return {};
629
630 append_utf8(result.second, (char32_t)parsed.second);
631 break;
632 }
633 case 'U':
634 {
635 auto num = consume_n(view, 8);
636 if (num.size() < 8 || view.empty()) return {};
637
638 auto parsed = consume_c_integer(num, 16);
639 if (parsed.first.empty() || !num.empty()) return {};
640
641 append_utf8(result.second, (char32_t)parsed.second);
642 break;
643 }
644 default:
645 return {};
646 }
647 }
648 else
649 {
650 result.second += cp;
651 }
652
653 if (view.empty())
654 return {};
655 }
656
657 if (!consume(view, DELIMITER))
658 return {};
659
660 result.first = make_sv(start, view.begin());
661 strv = view;
662 return result;
663 }

666 {
667 if (str.empty())
668 return {};
669

671 if (first_char == '-')
