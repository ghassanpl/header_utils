203 value eval_call(std::vector<value> args)
204 {
205 if (args.empty())
206 return rval(null_json);
207
208 //const auto orig_args = args;
209
210 std::string funcname;
211 std::vector<value> arguments;
212 if constexpr (decade_syntax)
213 {
214 const auto args_count = args.size();
215 const bool infix = (args_count % 2) == 1;
216
217 if (args_count >= 1 && (ref(args[0]) == "list" || ref(args[0]) == "eval"))
218 {
219 funcname = std::string{ ref(args[0]) } + ":";
220 arguments = std::move(args);
221 }
222 else if (args_count == 1)
223 {
224 funcname = ref(args[0]);
225 arguments = std::move(args);
226 }
227 else
228 {
229 arguments.push_back({});
230 if (infix)
231 {
232 arguments.push_back(std::move(args[0]));
233 funcname += ':';
234 }
235
236 bool first = true;
239 for (size_t i = infix; i < args_count; i += 2)
240 {
241 auto& function_identifier = args[i];
242 if (ref(function_identifier).is_string())
243 {
244 funcname += ref(function_identifier);
245 funcname += ':';
246
247 arguments.push_back(std::move(args[i + 1]));
248 }
249 else
250 return report_error("expected function name part, got: {}", ref(function_identifier).dump());
251 }
252
253 arguments[0] = rval(funcname);
254 }
255 }
256 else
257 {
258 auto& func = ref(args[0]);
259 if (func.is_string())
260 funcname = func.get_ref<json::string_t const&>();
261 else if (func.is_array())
262 {
263 auto eres = eval_args(func.get_ref<json::array_t const&>());
264 if (eres.is_string())
265 funcname = eres.get_ref<json::string_t const&>();
266 }
267 if (funcname.empty())
268 return report_error("first element of eval array must eval to a string func name, got: {}", func.dump());
269 }
270
271 if (eval_func const* func = find_func(funcname))
272 return (*func)(*this, std::move(arguments));
273 return report_error("func with name '{}' not found", funcname);
274 }

276 template <typename V>
277 value eval(V&& val)
278 {
279 if constexpr (std::same_as<V, value const&>)
280 {
281 return eval(value{ val });
282 }
283 else if constexpr (std::same_as<std::remove_cvref_t<V>, json>)
284 {
285 return eval(forward(std::forward<V>(val)));
286 }
287 else if constexpr (std::same_as<std::remove_cvref_t<V>, value>)
288 {
289 if (ref(val).is_string())
290 {
291 if (auto str = std::string_view{ ref(val) }; !str.empty())
292 {
293 for (auto& [prefix, macro] : prefix_macros)
294 {
295 if (str.starts_with(prefix))
296 {
297 return eval(macro(*this, { std::move(val) }));
298 }
299 }
300 }
301 }
302
303 if (!ref(val).is_array())
304 return std::move(val);
305
306 std::vector<value> args;
307 switch (val.index())
308 {
309 case 0:
310 {
311 json::array_t arr = std::move(std::get<json>(val).get_ref<json::array_t&>());
312 for (auto& a : arr)
313 args.push_back(rval(std::move(a)));
314 break;
315 }
316 case 1:
317 {
318 json::array_t& arr = std::get<json*>(val)->get_ref<json::array_t&>();
319 for (auto& a : arr)
320 args.push_back(rval(std::move(a)));
321 break;
322 }
323 case 2:
324 {
325 json::array_t const& arr = std::get<json const*>(val)->get_ref<json::array_t const&>();
326 for (auto& a : arr)
327 args.push_back(ref(a));
328 break;
329 }
330 }
331
332 return eval_call(std::move(args));
333 }

368 {
369 if (args.size() != arg_count + 1)
370 throw std::runtime_error(std::format("function {} requires exactly {} arguments, {} given", ref(args[0]).dump(), arg_count, args.size() - 1));

373 static void assert_args(std::span<value const> args, size_t min_args, size_t max_args)
374 {
375 if (args.size() < min_args + 1 && args.size() >= max_args + 1)
376 throw std::runtime_error(std::format("function {} requires between {} and {} arguments, {} given", ref(args[0]).dump(), min_args, max_args, args.size() - 1));
377 }

383 }
384
385 static auto assert_arg(std::span<value const> args, size_t arg_num, json::value_t type = json::value_t::discarded)
386 {
387 if (arg_num >= args.size())
388 throw std::runtime_error(std::format("function {} requires {} arguments, {} given", ref(args[0]).dump(), arg_num, args.size() - 1));
389
390 if (type != json::value_t::discarded && ref(args[arg_num]).type() != type)
391 {
392 throw std::runtime_error(std::format("argument #{} to function {} must be of type {}, {} given",
393 arg_num, ref(args[0]).dump(), detail::json_type_name(type), detail::json_type_name(ref(args[arg_num]).type())));
394 }
395
396 return ref(args[arg_num]).type();
397 }

399 template <std::same_as<nlohmann::json::value_t>... T>
400 static void assert_args(std::vector<value> args, T... arg_types)
401 {
402 static constexpr size_t arg_count = sizeof...(T);
403 assert_args(args, arg_count);
404
405 const auto types = std::array{ arg_types... };
406 for (size_t i = 0; i < types.size(); ++i)
407 {
408 if (types[i] != json::value_t::discarded)
409 assert_arg(args, i + 1, types[i]);

427 static json_pointer make_pointer(json const& index)
428 {
429 json_pointer ptr;
430 switch (index.type())
431 {
432 case string: ptr /= std::string{ index }; return ptr;
433 case number_float: ptr /= int(float(index)); return ptr;
434 case number_integer: ptr /= int(index); return ptr;
435 case number_unsigned: ptr /= int(index); return ptr;
436 case array: {
437 for (auto& e : index)
438 ptr /= make_pointer(e);
439 return ptr;
440 }
441 }
442 throw std::runtime_error(std::format("invalid value index type '{}'", index.type_name()));
443 }

477 static inline value while_do(env_type& e, std::vector<value> args)
478 {
479 e.assert_args(args, 2);
480 value last = null_json;
481 while (e.is_true(e.eval(args[1])))
482 {
483 try
484 {
485 last = e.eval(args[2]);
486 }
487 catch (e_break const& ex)
488 {
489 if (!e.ref(ex.result).is_discarded())
490 last = ex.result;
491 break;
492 }
493 catch (e_continue const& ex)
494 {
495 if (!e.ref(ex.result).is_discarded())
496 last = ex.result;
497 continue;
498 }
499 }
500 return last;
501 }
