264 template <typename FUNC>
265 constexpr auto for_each(FUNC&& callback) const
266 {
267 using return_type = std::invoke_result_t<FUNC, enum_type>;
268 using bitset_type = std::make_unsigned_t<value_type>;
269 auto bitset = static_cast<bitset_type>(bits);
270 while (bitset)
271 {
272#pragma warning(suppress: 4146)
273 const auto t = static_cast<bitset_type>(bitset & -bitset);
274 const auto r = std::countr_zero(t);
275 if constexpr (std::is_convertible_v<return_type, bool>)
276 {
277 if (auto ret = callback((enum_type)r)) return ret;
278 }
279 else
280 callback((enum_type)r);
281 bitset ^= t;
282 }
283 if constexpr (std::is_convertible_v<return_type, bool>)
284 return return_type{};
285 }

289 constexpr void apply(flag_changes changes) noexcept
290 {
291 const auto bits_to_xor = changes.bits[0].bits & changes.bits[1].bits;
292 bits ^= bits_to_xor;
293
295 changes.bits[0].bits &= ~bits_to_xor;
296 changes.bits[1].bits &= ~bits_to_xor;
297
299 bits &= ~changes.bits[1].bits;
300 bits |= changes.bits[0].bits;
301 }
