239 template <typename FUNC>
240 constexpr auto for_each(FUNC&& callback) const
241 {
242 using return_type = std::invoke_result_t<FUNC, enum_type>;
243 using bitset_type = std::make_unsigned_t<value_type>;
244 auto bitset = static_cast<bitset_type>(bits);
245 while (bitset)
246 {
247#pragma warning(suppress: 4146)
248 const auto t = static_cast<bitset_type>(bitset & -bitset);
249 const auto r = std::countr_zero(t);
250 if constexpr (std::is_convertible_v<return_type, bool>)
251 {
252 if (auto ret = callback((enum_type)r)) return ret;
253 }
254 else
255 callback((enum_type)r);
256 bitset ^= t;
257 }
258 if constexpr (std::is_convertible_v<return_type, bool>)
259 return return_type{};
260 }
