35 template <typename BUFFER, typename T>
36 bool buffer_append(BUFFER&& buffer, T&& val)
37 {
38 if constexpr (requires { buffer.append(val); })
39 {
40 buffer.append(val);
41 return true;
42 }
43 else if constexpr (requires { buffer.push_back(val); })
44 {
45 buffer.push_back(val);
46 return true;
47 }
48 else if constexpr (requires { buffer.insert(buffer.end(), val); })
49 {
50 buffer.insert(buffer.end(), val);
51 return true;
52 }
53 else if constexpr (requires { buffer.put(val); })
54 {
55 buffer.put(val);
56 return true;
57 }
58 else

60 static_assert(!std::same_as<std::void_t<BUFFER>, void>, "buffer cannot be appended with this value type - buffer_append might need to be specialized");

85 template <typename BUFFER, typename RANGE>

87 size_t buffer_append_range(BUFFER& buffer, RANGE&& range)
88 {
89 if constexpr (std::ranges::sized_range<RANGE>)
90 buffer_reserve(buffer, std::ranges::size(range));

126 template <typename BUFFER, typename ELEMENT_TYPE = buffer_element_type<BUFFER>>
127 requires output_buffer<BUFFER, ELEMENT_TYPE>
128 size_t buffer_append_utf8(BUFFER& buffer, char32_t cp)
129 {
130 size_t result = 0;
131 if (cp < 0x80)
132 {
133 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>(cp));
134 }
135 else if (cp < 0x800) {
136 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>((cp >> 6) | 0xc0));
137 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>((cp & 0x3f) | 0x80));
138 }
139 else if (cp < 0x10000) {
140 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>((cp >> 12) | 0xe0));
141 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>(((cp >> 6) & 0x3f) | 0x80));
142 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>((cp & 0x3f) | 0x80));
143 }
144 else {
145 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>((cp >> 18) | 0xf0));
146 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>(((cp >> 12) & 0x3f) | 0x80));
147 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>(((cp >> 6) & 0x3f) | 0x80));
148 result += buffer_append(buffer, static_cast<ELEMENT_TYPE>((cp & 0x3f) | 0x80));
149 }
150 return result;
151 }
