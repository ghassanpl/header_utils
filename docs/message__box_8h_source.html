141 template <typename... ARGS>
142 auto assumption_failure(std::string_view expectation, std::span<std::pair<std::string_view, std::string> const> values, std::string data, std::source_location loc, ARGS&&... args)
143 {
144 std::string message_information = std::format("Assumed: {}", expectation);
145 std::string message_long = std::format("In function `{}` at file `{}`, line {}\n\n", loc.function_name(), loc.file_name(), loc.line());
146 for (auto& [key, value] : values)
147 message_long += std::format("'{}' = '{}'\n", key, value);
148 if (!data.empty())
149 message_long += std::format("Additional Information: {}", move(data));
150 return ::ghassanpl::windows_message_box({
151 ::ghassanpl::msg::title{"Assumption Failed"},
152 ::ghassanpl::msg::description{message_information},
153 ::ghassanpl::msg::long_description{message_long},
154 ::ghassanpl::msg::debug_abort_continue_buttons,
155 1,
156 std::forward<ARGS>(args)...
157 });
158 }

300 inline windows_message_box_result windows_message_box(windows_message_box_params const& param)
301 {
302 using namespace win;
304 TASKDIALOGCONFIG task_config = {};
305 task_config.cbSize = sizeof(TASKDIALOGCONFIG);
306 task_config.hwndParent = param.window_handle ? (HWND)param.window_handle : GetActiveWindow();
307 task_config.hInstance = GetModuleHandleW(nullptr);
308 task_config.dwFlags =
309 /*TDF_ENABLE_HYPERLINKS*/ 0x0001
310 | /*TDF_USE_COMMAND_LINKS*/ 0x0010
311 | /*TDF_SIZE_TO_CONTENT*/ 0x01000000;
312 task_config.dwCommonButtons = 0;
313
315 task_config.cRadioButtons = 0;
316 task_config.pRadioButtons = nullptr;
317 task_config.nDefaultRadioButton = 0;
318
319 task_config.pszCollapsedControlText = nullptr;
320 task_config.hFooterIcon = nullptr;
321 task_config.pszFooter = nullptr;
322 task_config.cxWidth = 0;
323
325 auto title_u16 = string_ops::to_wstring(param.title);
326 auto description_u16 = string_ops::to_wstring(param.description);
327 auto long_description_u16 = string_ops::to_wstring(param.long_description);
328 auto checkbox_text_u16 = string_ops::to_wstring(param.checkbox_text);
329 auto additional_info_u16 = string_ops::to_wstring(param.additional_info);
330
331 task_config.pszWindowTitle = title_u16.c_str();
332 //AssumeBetween(int(icon), int(MessageBoxIcon::Warning), int(MessageBoxIcon::Security));
333#define MAKEINTRESOURCEW(i) ((LPWSTR)((ULONG_PTR)((WORD)(i))))
334 task_config.pszMainIcon = MAKEINTRESOURCEW(int(param.icon));
335#undef MAKEINTRESOURCEW
336 task_config.pszMainInstruction = description_u16.c_str();
337 if (!param.long_description.empty())
338 task_config.pszContent = long_description_u16.c_str();
339
340 if (!param.checkbox_text.empty())
341 task_config.pszVerificationText = checkbox_text_u16.c_str();
342 if (!param.additional_info.empty())
343 {
344 task_config.pszExpandedControlText = L"Additional Information";
345 task_config.pszExpandedInformation = additional_info_u16.c_str();
346 }
347
349 if (param.callback)
350 {
351 task_config.pfCallback = [](HWND, UINT uNotification, WPARAM wParam, LPARAM lParam, LONG_PTR dwRefData) -> HRESULT {
352 auto ptr = (decltype(param.callback)*)dwRefData;
353 return ptr->operator()((windows_message_box_event)uNotification, wParam, lParam) ? 0 : 1;
354 };
355 task_config.lpCallbackData = (LONG_PTR)&param.callback;
356 }
357
359 std::vector<TASKDIALOG_BUTTON> button_vector;
360 std::vector<std::wstring> button_texts;
361 std::ranges::transform(param.buttons, std::back_inserter(button_texts), string_ops::to_wstring);
362 std::ranges::transform(button_texts, std::back_inserter(button_vector), [id = 0](auto& wstr) mutable { return TASKDIALOG_BUTTON{ id++, wstr.c_str() }; });
363
364 task_config.cButtons = (UINT)button_vector.size();
365 task_config.pButtons = button_vector.data();
366 if (param.default_button_str.empty())
367 task_config.nDefaultButton = (int)param.default_button;
368 else
369 task_config.nDefaultButton = int(std::ranges::find(param.buttons, param.default_button_str) - param.buttons.begin());
370
371 int clicked_id = 0;
372 int checkbox_value = false;
373 InitCommonControls();
374
375 auto lib = win::LoadLibraryW(L"Comctl32.dll");
376 const auto TaskDialogIndirect = (TaskDialogIndirectPtr)win::GetProcAddress(lib, "TaskDialogIndirect");
377 auto result = TaskDialogIndirect(&task_config, &clicked_id, nullptr, &checkbox_value);
378 win::FreeLibrary(lib);
379
380 if (result != S_OK_)
381 {
382 return { true };
383 }
384
385 return { false, size_t(clicked_id), checkbox_value != 0 };
386 }
