139 template <typename... ARGS>
140 auto assumption_failure(std::string_view expectation, std::span<std::pair<std::string_view, std::string> const> values, std::string data, std::source_location loc, ARGS&&... args)
141 {
142 std::string message_information, message_long;
143 message_information += std::format("Assumed: {}", expectation);
144 message_long += std::format("In function `{}` at file `{}`, line {}\n\n", loc.function_name(), loc.file_name(), loc.line());
145 for (auto& kvp : values)
146 message_long += std::format("'{}' = '{}'\n", kvp.first, kvp.second);
147 if (!data.empty())
148 message_long += std::format("Additional Information: {}", move(data));
149 return ::ghassanpl::windows_message_box({
150 ::ghassanpl::msg::title{"Assumption Failed"},
151 ::ghassanpl::msg::description{message_information},
152 ::ghassanpl::msg::long_description{message_long},
153 ::ghassanpl::msg::debug_abort_continue_buttons,
154 1,
155 std::forward<ARGS>(args)...
156 });
157 }

299 inline windows_message_box_result windows_message_box(windows_message_box_params const& param)
300 {
301 using namespace win;
303 TASKDIALOGCONFIG task_config = {};
304 task_config.cbSize = sizeof(TASKDIALOGCONFIG);
305 task_config.hwndParent = param.window_handle ? (HWND)param.window_handle : GetActiveWindow();
306 task_config.hInstance = GetModuleHandleW(nullptr);
307 task_config.dwFlags =
308 /*TDF_ENABLE_HYPERLINKS*/ 0x0001
309 | /*TDF_USE_COMMAND_LINKS*/ 0x0010
310 | /*TDF_SIZE_TO_CONTENT*/ 0x01000000;
311 task_config.dwCommonButtons = 0;
312
314 task_config.cRadioButtons = 0;
315 task_config.pRadioButtons = nullptr;
316 task_config.nDefaultRadioButton = 0;
317
318 task_config.pszCollapsedControlText = nullptr;
319 task_config.hFooterIcon = nullptr;
320 task_config.pszFooter = nullptr;
321 task_config.cxWidth = 0;
322
324 auto title_u16 = string_ops::to_wstring(param.title);
325 auto description_u16 = string_ops::to_wstring(param.description);
326 auto long_description_u16 = string_ops::to_wstring(param.long_description);
327 auto checkbox_text_u16 = string_ops::to_wstring(param.checkbox_text);
328 auto additional_info_u16 = string_ops::to_wstring(param.additional_info);
329
330 task_config.pszWindowTitle = title_u16.c_str();
331 //AssumeBetween(int(icon), int(MessageBoxIcon::Warning), int(MessageBoxIcon::Security));
332#define MAKEINTRESOURCEW(i) ((LPWSTR)((ULONG_PTR)((WORD)(i))))
333 task_config.pszMainIcon = MAKEINTRESOURCEW(int(param.icon));
334#undef MAKEINTRESOURCEW
335 task_config.pszMainInstruction = description_u16.c_str();
336 if (!param.long_description.empty())
337 task_config.pszContent = long_description_u16.c_str();
338
339 if (!param.checkbox_text.empty())
340 task_config.pszVerificationText = checkbox_text_u16.c_str();
341 if (!param.additional_info.empty())
342 {
343 task_config.pszExpandedControlText = L"Additional Information";
344 task_config.pszExpandedInformation = additional_info_u16.c_str();
345 }
346
348 if (param.callback)
349 {
350 task_config.pfCallback = [](HWND hwnd, UINT uNotification, WPARAM wParam, LPARAM lParam, LONG_PTR dwRefData) {
351 auto ptr = (decltype(param.callback)*)dwRefData;
352 return ptr->operator()((windows_message_box_event)uNotification, wParam, lParam) ? S_OK : S_FALSE;
353 };
354 task_config.lpCallbackData = (LONG_PTR)&param.callback;
355 }
356
358 std::vector<TASKDIALOG_BUTTON> button_vector;
359 std::vector<std::wstring> button_texts;
360 std::transform(param.buttons.begin(), param.buttons.end(), std::back_inserter(button_texts), string_ops::to_wstring);
361 std::transform(button_texts.begin(), button_texts.end(), std::back_inserter(button_vector), [id = 0](auto& wstr) mutable { return TASKDIALOG_BUTTON{ id++, wstr.c_str() }; });
362
363 task_config.cButtons = (UINT)button_vector.size();
364 task_config.pButtons = button_vector.data();
365 if (param.default_button_str.empty())
366 task_config.nDefaultButton = (int)param.default_button;
367 else
368 task_config.nDefaultButton = int(std::find(param.buttons.begin(), param.buttons.end(), param.default_button_str) - param.buttons.begin());
369
370 int clicked_id = 0;
371 int checkbox_value = false;
372 InitCommonControls();
373 auto result = TaskDialogIndirect(&task_config, &clicked_id, nullptr, &checkbox_value);
374 if (result != ((HRESULT)0L))
375 {
376 return { true };
377 }
378
379 return{ false, size_t(clicked_id), checkbox_value != 0 };
380 }
