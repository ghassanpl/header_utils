75 inline std::string parse_string_value(std::string_view& str)
76 {
77 string_ops::trim_whitespace_left(str);
78 auto first = str[0];
79 if (first == '\'' || first == '"')
80 return parse_literal(str);
81 else if (string_ops::ascii::isalpha(first) || first == '_')
82 return parse_identifier(str);
83
84 return {};

87 inline nlohmann::json parse_string(std::string_view& str)
88 {
89 if (string_ops::ascii::isalpha(str[0]))

91 auto string = parse_string_value(str);

94 else if (string == "false")
95 return false;
96 else if (string == "null" || string == "nil")
97 return nullptr;
98 return string;

102 return parse_string_value(str);

106 inline nlohmann::json parse_object(std::string_view& str)
107 {
108 auto obj = nlohmann::json::object();
109
110 do
111 {
112 string_ops::trim_whitespace_left(str);
113 if (string_ops::consume(str, '}'))
114 return obj;
115
116 auto key = parse_string_value(str);
117 string_ops::trim_whitespace_left(str);
118
119 if (str.empty())
120 break;
121
122 if (str[0] == ',' || str[0] == ';')
123 {
124 obj[std::move(key)] = true;
125 }
126 else
127 {
128 if (str[0] == '=' || str[0] == ':')
129 str.remove_prefix(1);
130
131 string_ops::trim_whitespace_left(str);
132
133 auto val = parse(str);
134 obj[std::move(key)] = std::move(val);
135 }
136
137 string_ops::trim_whitespace_left(str);
138
139 if (!str.empty() && (str[0] == ',' || str[0] == ';'))
140 str.remove_prefix(1);
141 } while (!str.empty());
142
143 return obj;
144 }

146 inline nlohmann::json parse(std::string_view& str)
147 {
148 string_ops::trim_whitespace_left(str);
149 if (str.empty()) return {};
150
151 auto first = str[0];
152 if (string_ops::consume(str, '{'))
153 return parse_object(str);
154 else if (string_ops::consume(str, '[') || string_ops::consume(str, '('))
155 return parse_array(str);
156 else if (string_ops::ascii::isalpha(first) || first == '\'' || first == '"')
157 return parse_string(str);
158 else if (string_ops::ascii::isdigit(first) || first == '-')
159 {
160 double result = 0;
161 auto fcresult = std::from_chars(str.data(), str.data() + str.size(), result);
162 str = string_ops::make_sv(fcresult.ptr, str.end());
163 return result;
164 }
165 return {};
166 }

176 template <typename OUTFUNC>
177 void output_string(OUTFUNC&& func, std::string_view strval)
178 {
179 func("\"");
180 std::string buf;
181 char temp[8]{};
182 auto start = strval.begin();
183 for (auto it = strval.begin(); it != strval.end(); ++it)
184 {
185 if (string_ops::ascii::isprint(*it) && *it != '"' && *it != '\\')
186 continue;
187 else
188 {
189 if (start != it)
190 func(string_ops::make_sv(start, it));
191
192 func("\\x");
193 auto&& [end, ec] = std::to_chars(std::begin(temp), std::end(temp), static_cast<uint8_t>(*it), 16);
194 func(string_ops::make_sv(std::begin(temp), end));
195
196 start = std::next(it);
197 }
198 }
199 if (start != strval.end())
200 func(string_ops::make_sv(start, strval.end()));
201 func("\"");
202 }

220 template <typename OUTFUNC>
221 void output(OUTFUNC&& out, nlohmann::json const& value)
222 {
223 switch (value.type())
224 {
225 case nlohmann::json::value_t::null: out("null"); return;
226 case nlohmann::json::value_t::object:
227 out("{ ");
228 for (auto& obj : value.get_ref<nlohmann::json::object_t const&>())
229 {
230 wilson::detail::output_string(out, obj.first);
231 out(": ");
232 wilson::output(out, obj.second);
233 out(", ");
234 }
235 out(" }");
236 return;
237 case nlohmann::json::value_t::array:
238 out("[ ");
239 for (auto& obj : value.get_ref<nlohmann::json::array_t const&>())
240 {
241 wilson::output(out, obj);
242 out(", ");
243 }
244 out(" ]");
245 return;
246 case nlohmann::json::value_t::string: wilson::detail::output_string(out, value.get_ref<nlohmann::json::string_t const&>()); return;
247 case nlohmann::json::value_t::boolean: out(value.get_ref<nlohmann::json::boolean_t const&>() ? "true" : "false"); return;
248 case nlohmann::json::value_t::number_integer: wilson::detail::output_value(out, value.get_ref<nlohmann::json::number_integer_t const&>()); return;
249 case nlohmann::json::value_t::number_unsigned: wilson::detail::output_value(out, value.get_ref<nlohmann::json::number_unsigned_t const&>()); return;
250 case nlohmann::json::value_t::number_float: wilson::detail::output_value(out, value.get_ref<nlohmann::json::number_float_t const&>()); return;
251 case nlohmann::json::value_t::binary:
252 out("[ ");
253 for (auto byte : value.get_ref<nlohmann::json::binary_t const&>())
254 {
255 wilson::detail::output_value(out, byte);
256 out(", ");
257 }
258 out(" ]");
259 return;
260 case nlohmann::json::value_t::discarded: return;
261 }
262 }
