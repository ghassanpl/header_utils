200 constexpr inline char32_t last_unicode_code_point = 0x10FFFF;
201 constexpr inline char32_t first_unicode_high_surrogate = 0xD800;
202 constexpr inline char32_t last_unicode_high_surrogate = 0xDBFF;
203 constexpr inline char32_t first_unicode_low_surrogate = 0xDC00;
204 constexpr inline char32_t last_unicode_low_surrogate = 0xDFFF;
205
206 constexpr bool is_high_surrogate(char32_t cp) noexcept { return cp >= first_unicode_high_surrogate && cp <= last_unicode_high_surrogate; }
207 constexpr bool is_low_surrogate(char32_t cp) noexcept { return cp >= first_unicode_low_surrogate && cp <= last_unicode_low_surrogate; }
208 constexpr bool is_surrogate(char32_t cp) noexcept { return cp >= first_unicode_high_surrogate && cp <= last_unicode_low_surrogate; }
209 constexpr bool is_unicode(char32_t cp) noexcept { return cp <= last_unicode_code_point; }
210 constexpr auto get_unicode_plane(char32_t cp) noexcept -> unicode_plane { return is_unicode(cp) ? unicode_plane(cp >> 16) : unicode_plane::invalid; }
211 constexpr bool is_unicode_character(char32_t cp) noexcept { return is_unicode(cp) && ((cp & 0xFFFE) != 0xFFFE) && !(cp >= 0xFDD0 && cp <= 0xFDEF); }

213 constexpr char32_t surrogate_pair_to_codepoint(char32_t high, char32_t low) noexcept
214 {
215 return 0x10000 + ((high - first_unicode_high_surrogate) << 10) + (low - first_unicode_low_surrogate);
216 }
217
218 constexpr std::pair<char32_t, char32_t> codepoint_to_surrogate_pair(char32_t cp) noexcept
219 {
220 return { ((cp - 0x10000) >> 10) + first_unicode_high_surrogate, ((cp - 0x10000) & 0x3FF) + first_unicode_low_surrogate };
221 }

242 inline text_decode_result decode_codepoint(stringable8 auto _str, text_encoding encoding)
243 {
244 auto sv = make_sv(_str);
245 if (sv.empty())
246 return { text_decode_result::truncated };
247
248 switch (encoding.base_encoding)
249 {
250 case base_text_encoding::utf8:
251 {
252 const auto first = std::bit_cast<uint8_t>(sv[0]);
253 if (first < 0x80)
254 return { text_decode_result::valid, first, 1 };
255
256 char32_t value = 0;
257 uint8_t length = 0;
258 if ((first >> 5) == 0x6) { length = 2; value = first & 0x1F; }
259 else if ((first >> 4) == 0xe) { length = 3; value = first & 0xF; }
260 else if ((first >> 3) == 0x1e) { length = 4; value = first & 0x7; }
261
262 if (length == 0 || sv.size() < length)
263 return { text_decode_result::truncated, first, 1 };
264
265 switch (length)
266 {
267 case 4:
268 if ((sv[3] & 0xC0) != 0x80)
269 return { text_decode_result::invalid, first, 1 };
270 value = (value << 6) | (sv[3] & 0x3f);
271 case 3:
272 if ((sv[2] & 0xC0) != 0x80)
273 return { text_decode_result::invalid, first, 1 };
274 value = (value << 6) | (sv[2] & 0x3f);
275 case 2:
276 if ((sv[1] & 0xC0) != 0x80)
277 return { text_decode_result::invalid, first, 1 };
278 value = (value << 6) | (sv[1] & 0x3f);
279 return { text_decode_result::valid, value, length };
280 }
281 }
282 case base_text_encoding::utf16:
283 {
284 if (sv.size() < 2)
285 return { text_decode_result::truncated };
286
287 const uint16_t first = detail::get_val<uint16_t>(sv.data(), encoding.endianness);
288 if (is_surrogate(first))
289 {
290 if (is_high_surrogate(first))
291 {
292 if (sv.size() < 4)
293 return { text_decode_result::truncated, first, 2 };
294
295 const uint16_t second = detail::get_val<uint16_t>(sv.data() + 2, encoding.endianness);
296 if (is_low_surrogate(second))
297 return { text_decode_result::valid, surrogate_pair_to_codepoint(first, second), 4 };
298 }
299
300 return { text_decode_result::invalid, first, 2 };
301 }
302
303 return { text_decode_result::valid, first, 2 };
304 }
305 case base_text_encoding::utf32:
306 {
307 if (sv.size() < 4)
308 return { text_decode_result::truncated };
309
310 const uint32_t cp = detail::get_val<uint32_t>(sv.data(), encoding.endianness);
311 if (is_unicode(cp))
312 return { text_decode_result::valid, cp, 4 };
313 return { text_decode_result::invalid, cp, 4 };
314 }
315 default:
316 return {};
317 }
318 }

320 inline text_encoding detect_encoding(stringable8 auto str)
321 {
322 struct TextFileStats {
323 size_t points = 0;
324 size_t valid_points = 0;
325 size_t control_points = 0;
326 size_t plain_ascii = 0;
327 size_t whitespace = 0;
328 size_t extended_codepoints = 0;
329 float one_over_points = 0.f;
330 size_t invalid_points() const { return points - valid_points; }
331 float score() const { return (2.5f * whitespace + plain_ascii - 100.f * invalid_points() - 50.f * control_points + 5.f * extended_codepoints) * one_over_points; }
332 };
333
334 static const auto scanTextFile = [](TextFileStats& stats, stringable8 auto sv, text_encoding encoding) {
335 size_t numBytes = 0;
336 while (!sv.empty())
337 {
338 text_decode_result decoded = decode_codepoint(sv, encoding);
339 if (decoded.status == text_decode_result::truncated)
340 break;
341 sv.remove_prefix(decoded.byte_count);
342 numBytes += decoded.byte_count;
343 stats.points++;
344 if (decoded.status == text_decode_result::valid)
345 {
346 stats.valid_points++;
347 if (decoded.point < 32)
348 {
349 if (decoded.point == '\n' || decoded.point == '\t')
350 {
351 stats.plain_ascii++;
352 stats.whitespace++;
353 }
354 else if (decoded.point == '\r')
355 stats.plain_ascii++;
356 else
357 stats.control_points++;
358 }
359 else if (decoded.point < 127)
360 {
361 stats.plain_ascii++;
362 if (decoded.point == ' ')
363 stats.whitespace++;
364 }
365 else if (decoded.point >= 65536)
366 stats.extended_codepoints++;
367 }
368 }
369 if (stats.points > 0)
370 stats.one_over_points = 1.f / stats.points;
371 return numBytes;
372 };
373
374 auto sv = make_sv(str);
375
376 if (sv.empty())
377 return unknown_text_encoding;
378
379 if (text_encoding encoding = consume_bom(sv); encoding != unknown_text_encoding)
380 return encoding;
381
382 sv = sv.substr(0, 4096);
383
384 TextFileStats stats8;
385
387 size_t numBytesRead = scanTextFile(stats8, sv, utf8_encoding);
388 if (numBytesRead == 0)
389 return utf8_encoding;
390
392 if (stats8.invalid_points() == 0 && stats8.control_points == 0)
393 return utf8_encoding;
394
397 text_encoding encoding8 = utf8_encoding;
398 if (const auto numHighBytes = numBytesRead - stats8.plain_ascii - stats8.control_points;
399 stats8.invalid_points() >= numHighBytes * 0.2f)
400 {
402 encoding8 = unknown_text_encoding;
403 stats8.points = numBytesRead;
404 stats8.valid_points = numBytesRead;
405 }
406
408 TextFileStats stats16_le;
409 scanTextFile(stats16_le, sv, utf16_le_encoding);
410
411 TextFileStats stats16_be;
412 scanTextFile(stats16_be, sv, utf16_be_encoding);
413
415 TextFileStats* stats16 = &stats16_le;
416 text_encoding encoding16 = utf16_le_encoding;
417 if (stats16_be.score() > stats16_le.score())
418 {
419 stats16 = &stats16_be;
420 encoding16 = utf16_be_encoding;
421 }
422
424 TextFileStats stats32_le;
425 scanTextFile(stats32_le, sv, utf32_le_encoding);
426
427 TextFileStats stats32_be;
428 scanTextFile(stats32_be, sv, utf32_be_encoding);
429
431 TextFileStats* stats32 = &stats32_le;
432 text_encoding encoding32 = utf32_le_encoding;
433 if (stats32_be.score() > stats32_le.score())
434 {
435 stats32 = &stats32_be;
436 encoding32 = utf32_be_encoding;
437 }
438
440 const auto score8 = stats8.score();
441 const auto score16 = stats16->score();
442 const auto score32 = stats32->score();
443 if (score8 >= score32)
444 {
445 if (score16 >= score8)
446 return encoding16;
447 return encoding8;
448 }
449
450 if (score32 >= score16)
451 return encoding32;
452 return encoding16;
453 }

462 [[nodiscard]] constexpr char32_t consume_utf8(string_view8 auto& str)
463 {
464 using char_type = typename std::remove_cvref_t<decltype(str)>::value_type;
465 using unsigned_char_type = std::make_unsigned_t<char_type>;
466
467 if (str.empty()) return 0;
468 auto it = std::to_address(str.begin());
469 char32_t cp = static_cast<unsigned_char_type>(*it);
470
471 int length = 0;
472 if (cp < 0x80) length = 1;
473 else if ((cp >> 5) == 0x6) length = 2;
474 else if ((cp >> 4) == 0xe) length = 3;
475 else if ((cp >> 3) == 0x1e) length = 4;
476 else return 0;
477
478 switch (length) {
479 case 2:
480 ++it; cp = ((cp << 6) & 0x7ff) + (static_cast<unsigned_char_type>(*it) & 0x3f);
481 break;
482 case 3:
483 ++it; cp = ((cp << 12) & 0xffff) + ((static_cast<unsigned_char_type>(*it) << 6) & 0xfff);
484 ++it; cp += static_cast<unsigned_char_type>(*it) & 0x3f;
485 break;
486 case 4:
487 ++it; cp = ((cp << 18) & 0x1fffff) + ((static_cast<unsigned_char_type>(*it) << 12) & 0x3ffff);
488 ++it; cp += (static_cast<unsigned_char_type>(*it) << 6) & 0xfff;
489 ++it; cp += static_cast<unsigned_char_type>(*it) & 0x3f;
490 break;
491 }
492 str.remove_prefix(length);
493 return cp;
494 }

550 constexpr void transcode_codepage_to_utf8(string8 auto& dest, stringable8 auto source, std::span<char32_t const, 128> codepage_map)
551 {
552 using dest_char = typename std::decay_t<decltype(dest)>::value_type;
553 for (uint8_t cp : source)
554 {
555 if (cp < 0x80)
556 dest += static_cast<dest_char>(cp);
557 else
558 append_utf8(dest, codepage_map[cp - 0x80]);
559 }
