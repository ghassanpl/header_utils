165 constexpr inline color_t to_rgb(color_hsva_t const& hsva)
166 {
167 auto hue = hsva->r, saturation = hsva->g, value = hsva->b, alpha = hsva->a;
168 const int i = (int)hue;
169 float fraction = hue - (float)i;
170 if (!(i & 1)) fraction = 1.0f - fraction;
171 float m = value * (1.0f - saturation);
172 float n = value * (1.0f - saturation * fraction);
173 switch (i)
174 {
175 case 6:
176 case 0: return color_t(value, n, m, alpha);
177 case 1: return color_t(n, value, m, alpha);
178 case 2: return color_t(m, value, n, alpha);
179 case 3: return color_t(m, n, value, alpha);
180 case 4: return color_t(n, m, value, alpha);
181 case 5: return color_t(value, m, n, alpha);
182 default: return color_t(value, value, value, alpha);
183 }
184 }

187 constexpr inline color_hsva_t to_hsv(color_t const& rgba)
188 {
189 auto min = detail::min3(rgba.r, rgba.g, rgba.b);
190 auto max = detail::max3(rgba.r, rgba.g, rgba.b);
191 auto delta = max - min;
192
193 float h = 0;
194
195 if (delta != 0)
196 {
197 if (rgba.r == max)
198 h = (rgba.g - rgba.b) / delta;
199 else if (rgba.g == max)
200 h = 2 + (rgba.b - rgba.r) / delta;
201 else
202 h = 4 + (rgba.r - rgba.g) / delta;
203 h /= 6.0;
204 if (h < 0)
205 h += 1.0;
206 }
207
208 return color_hsva_t{
209 {
210 h,
211 (max != 0) ? (delta / max) : 0,
212 max,
213 rgba.a
214 }
215 };
216 }

219 inline constexpr color_rgba_t operator ""_rgb(const char* str, size_t n)
220 {
221 if (n == 0)
222 throw n;
223 if (str[0] == '#')
224 {
225 str++;
226 n--;
227 }
228 uint8_t r{}, g{}, b{}, a = 255;
229 switch (n)
230 {
231 case 4:
232 std::from_chars(str + 3, str + 4, a, 16); a = a << 4 | a;
233 case 3:
234 std::from_chars(str + 0, str + 1, r, 16); r = r << 4 | r;
235 std::from_chars(str + 1, str + 2, g, 16); g = g << 4 | g;
236 std::from_chars(str + 2, str + 3, b, 16); b = b << 4 | b;
237 break;
238 case 8:
239 std::from_chars(str + 6, str + 8, a, 16);
240 case 6:
241 std::from_chars(str + 0, str + 2, r, 16);
242 std::from_chars(str + 2, str + 4, g, 16);
243 std::from_chars(str + 4, str + 6, b, 16);
244 break;
245 default: throw "invalid number of characters";
246 }
247 return { r / 255.0f, g / 255.0f, b / 255.0f, a / 255.0f };
