144 {

146
147 virtual uri_error validate(uri_view uri) const noexcept
148 {
149 auto decomposed = decompose_uri(uri);
150 if (!decomposed) return decomposed.error();
151
152 if (decomposed.value().scheme != scheme()) return uri_error_code::scheme_invalid;
153
154 if (auto err = validate_authority(decomposed.value().authority)) return err;
155 if (auto err = validate_path(decomposed.value().path)) return err;
156 if (auto err = validate_query(decomposed.value().query)) return err;
157 if (auto err = validate_fragment(decomposed.value().fragment)) return err;
158
159 return {};
160 }
